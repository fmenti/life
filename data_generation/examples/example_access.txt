import pyvo as vo

adql_query="""
    SELECT TOP 10 object_id, main_id FROM life_td.object
    WHERE type='st'"""

link='http://dc.zah.uni-heidelberg.de/tap'
service = vo.dal.TAPService(link)

result=service.run_async(adql_query.format(**locals()), 
                                             maxrec=160000)
    
catalog=result.to_table()
